 name: ssd-monitor

   matchLabels:
     app: ssd-monitor

   metadata:
     labels:
       app: ssd-monitor

Rediness
======== 
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: readiness
  name: readiness
spec:
  replicas: 3
  selector:
    matchLabels:
      app: readiness
  template:
    metadata:
      labels:
        app: readiness
    spec:
      containers:
      - image: devopstechlab1/kubweb
        name: readiness
        ports:
        - containerPort: 80
        readinessProbe:
          httpGet:
            path: /
            port: 80
          initialDelaySeconds: 15
